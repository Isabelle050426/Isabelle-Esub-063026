                    mimeType = 'text/html';
                    break;
                case 'pdf':

                    alert('系統正在為您調用底層原生列印引擎與虛擬 PDF 印表機轉換模組，請於彈出視窗中選擇「另存為 PDF」。');
                    window.print();
                    return;
            }

            const blob = new Blob([content], { type: mimeType });
            const link = document.createElement('a');
            link.href = URL.createObjectURL(blob);
            link.download = filename;
            link.click();
